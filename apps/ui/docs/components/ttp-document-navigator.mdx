## Usage

```tsx
import { TtpDocumentNavigator } from '@/components/ui/totheprod-ui/ttp-document-navigator'
import type { DocumentSection } from '@/components/ui/totheprod-ui/ttp-document-navigator'

const sections: DocumentSection[] = [
  {
    id: "workspace",
    title: "Workspace",
    isCollapsible: true,
    defaultCollapsed: false,
    pages: [
      {
        id: "page-1",
        title: "Getting Started",
        href: "/getting-started",
        isActive: true,
      },
      {
        id: "page-2",
        title: "Project Planning",
        href: "/project-planning",
        children: [
          {
            id: "page-2-1",
            title: "Sprint 1",
            href: "/sprint-1",
          },
        ],
      },
    ],
  },
]

export const App = () => {
  return <TtpDocumentNavigator sections={sections} />
}
```

| --------------- | ------------ | ------- | ---------------------------------------------- |

| isCollapsible   | boolean      | -       | Whether the section can be collapsed           |

## Examples

### Basic Usage

```tsx
<TtpDocumentNavigator
  sections={[
    {
      id: "main",
      pages: [
        { id: "1", title: "Home", href: "/" },
        { id: "2", title: "About", href: "/about" },
      ],
    },
  ]}
/>
```

### With Nested Pages

```tsx
<TtpDocumentNavigator
  sections={[
    {
      id: "workspace",
      title: "Workspace",
      pages: [
        {
          id: "project",
          title: "Project",
          children: [
            { id: "sprint-1", title: "Sprint 1", href: "/sprint-1" },
            { id: "sprint-2", title: "Sprint 2", href: "/sprint-2" },
          ],
        },
      ],
    },
  ]}
/>
```

### Custom Styling

```tsx
<TtpDocumentNavigator
  sections={sections}
  width="280px"
  className="border-r-2 border-primary"
  showSectionTitles={false}
/>
```

## Best Practices

> **Tip**: Use meaningful IDs and titles for better organization and accessibility.

- Keep section titles concise
- Limit nesting depth to 3-4 levels for better UX
- Use `isActive` prop to indicate the current page
- Consider using custom icons for important pages
- Use collapsible sections for better organization of large navigation structures
