## Usage

### Basic Example

```tsx
'use client'

import { useState } from 'react'
import {
  CommandPalette,
  useCommandPalette,
  type RaycastSection,
} from '@/components/ui/totheprod-ui/ttp-command-palette'
import { FolderOpen, Menu, MessageCircle } from 'lucide-react'

export const Example = () => {
  const { open, setOpen } = useCommandPalette()

  const sections: RaycastSection[] = [
    {
      id: 'favorites',
      title: 'Favorites',
      items: [
        {
          id: 'recent-projects',
          icon: <FolderOpen className="size-5" />,
          title: 'Search Recent Projects',
          description: 'Cursor',
          type: 'Command',
          onSelect: () => console.log('Selected!'),
        },
      ],
    },
  ]

  return (
    <>
      <button onClick={() => setOpen(true)}>Open Menu</button>
      <CommandPalette open={open} onOpenChange={setOpen} sections={sections} />
    </>
  )
}
```

### With Actions

Items can have contextual actions that appear when selected:

```tsx
const sections: RaycastSection[] = [
  {
    id: 'favorites',
    title: 'Favorites',
    items: [
      {
        id: 'project',
        icon: <FolderOpen className="size-5" />,
        title: 'My Project',
        type: 'Command',
        actions: [
          {
            id: 'open',
            label: 'Open Command',
            shortcut: ['‚Üµ'],
            onAction: () => console.log('Opening...'),
          },
          {
            id: 'remove',
            label: 'Remove from Favorites',
            shortcut: ['‚å•', '‚åò', 'F'],
            onAction: () => console.log('Removed'),
          },
        ],
      },
    ],
  },
]
```

### With Banner

Display announcements or updates at the top:

```tsx
<CommandPalette
  open={open}
  onOpenChange={setOpen}
  sections={sections}
  banner={{
    icon: <Sparkles className="size-5 text-red-500" />,
    title: "See what's new?",
    version: 'v1.103.6',
    action: {
      label: 'Update',
      onClick: () => console.log('Updating...'),
    },
  }}
/>
```

## API Reference

### CommandPaletteProps

| Prop            | Type                      | Default                             | Description                     |
| --------------- | ------------------------- | ----------------------------------- | ------------------------------- |
| `open`          | `boolean`                 | -                                   | Controls menu visibility        |
| `onOpenChange`  | `(open: boolean) => void` | -                                   | Callback when menu opens/closes |
| `sections`      | `RaycastSection[]`        | -                                   | Menu sections and items         |
| `placeholder`   | `string`                  | `'Search for apps and commands...'` | Search input placeholder        |
| `askAiLabel`    | `string`                  | `'Ask AI'`                          | Label for AI button             |
| `emptyText`     | `string`                  | `'No results found.'`               | Empty state text                |
| `showFooter`    | `boolean`                 | `true`                              | Show/hide footer                |
| `footerActions` | `object`                  | -                                   | Custom footer action labels     |
| `banner`        | `object`                  | -                                   | Optional banner configuration   |

### RaycastSection

```ts
type RaycastSection = {
  id: string
  title: string
  items: RaycastCommandItem[]
}
```

### RaycastCommandItem

```ts
type RaycastCommandItem = {
  id: string
  icon: React.ReactNode
  title: string
  description?: string
  type?: string
  onSelect?: () => void
  actions?: RaycastAction[]
}
```

### RaycastAction

```ts
type RaycastAction = {
  id: string
  icon?: React.ReactNode
  label: string
  shortcut?: string[]
  onAction?: () => void
}
```

## Hook: useCommandPalette

A convenient hook that sets up keyboard shortcuts (‚åòK / Ctrl+K):

```tsx
const { open, setOpen } = useCommandPalette()
```

## Keyboard Shortcuts

- `‚åòK` or `Ctrl+K` - Toggle menu (when using the hook)
- `‚Üë` / `‚Üì` - Navigate items
- `‚Üµ` - Select item
- `Esc` - Close menu
- `Tab` - Access AI features
